```bash
# 克隆项目
git clone -b vue3.0-antdv https://github.com/chuzhixin/vue-admin-beautiful.git
# 进入项目目录
cd vue-admin-beautiful-pro
# 安装依赖
npm i
# 本地开发 启动项目
npm run serve
```
